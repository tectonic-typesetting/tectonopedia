% Creating hyperlinks.
%
\RequirePackage{hyperref}
%
% \href{URL}{TEXT}
%  This builds in a `target="_blank"` setting.
\renewcommand{\href}[2]{%
  \special{tdux:mfs a^^J%
Dtarget _blank^^J%
Dhref #1}#2\special{tdux:me a}%
}
%
% \hrefInternal{URL}{TEXT}
%  Like \href, but without the "_blank" target.
\newcommand{\hrefInternal}[2]{%
  \special{tdux:mfs a^^J%
Dhref #1}#2\special{tdux:me a}%
}
